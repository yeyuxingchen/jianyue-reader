```typescript
// src/services/aiService.ts
export function updateCSPForAI(baseUrl: string): void {
  try {
    const domain = new URL(baseUrl).origin
    const meta = document.querySelector('meta[http-equiv="Content-Security-Policy"]')
    if (!meta) return
    
    const current = meta.getAttribute('content') || ''
    // 只更新 connect-src 部分
    const updated = current.replace(
      /connect-src[^;]+;/,
      `connect-src 'self' ${domain} blob: data:;`
    )
    meta.setAttribute('content', updated)
  } catch (err) {
    console.error('Failed to update CSP:', err)
  }
}
```

### 1.2 文件路径验证

#### 完整实现

```typescript
// electron/security.ts
import { app } from 'electron'
import path from 'path'
import fs from 'fs'

// 允许访问的目录列表
const DEFAULT_ALLOWED_DIRS = [
  'userData',
  'documents',
  'downloads',
  'desktop',
  'temp',
]

let allowedDirs: string[] = []

/**
 * 初始化安全模块
 */
export function initSecurity(): void {
  allowedDirs = DEFAULT_ALLOWED_DIRS.map(dir => {
    try {
      return app.getPath(dir as any)
    } catch {
      return null
    }
  }).filter(Boolean) as string[]
  
  // 添加用户自定义目录
  const userDirs = getUserAuthorizedDirs()
  allowedDirs.push(...userDirs)
}

/**
 * 获取用户授权的目录
 */
function getUserAuthorizedDirs(): string[] {
  const configPath = path.join(app.getPath('userData'), 'authorized-dirs.json')
  try {
    if (fs.existsSync(configPath)) {
      const data = JSON.parse(fs.readFileSync(configPath, 'utf-8'))
      return data.dirs || []
    }
  } catch {
    // 配置文件不存在或格式错误
  }
  return []
}

/**
 * 保存用户授权的目录
 */
export function saveUserAuthorizedDirs(dirs: string[]): void {
  const configPath = path.join(app.getPath('userData'), 'authorized-dirs.json')
  fs.writeFileSync(configPath, JSON.stringify({ dirs }, null, 2))
}

/**
 * 验证文件路径是否在允许范围内
 */
export function isPathAllowed(filePath: string): boolean {
  if (!filePath) return false
  
  const resolvedPath = path.resolve(filePath)
  
  // 检查是否在允许的目录下
  return allowedDirs.some(dir => {
    const resolvedDir = path.resolve(dir)
    return resolvedPath.startsWith(resolvedDir + path.sep) || resolvedPath === resolvedDir
  })
}

/**
 * 验证并规范化文件路径
 * @throws {Error} 如果路径不被允许
 */
export function validateFilePath(filePath: string): string {
  if (!isPathAllowed(filePath)) {
    throw new Error(`Access denied: ${filePath}`)
  }
  return path.resolve(filePath)
}

/**
 * 添加授权目录
 */
export function addAuthorizedDir(dirPath: string): boolean {
  const resolvedPath = path.resolve(dirPath)
  
  // 验证目录存在
  if (!fs.existsSync(resolvedPath) || !fs.statSync(resolvedPath).isDirectory()) {
    return false
  }
  
  // 检查是否已存在
  if (allowedDirs.includes(resolvedPath)) {
    return true
  }
  
  allowedDirs.push(resolvedPath)
  saveUserAuthorizedDirs(allowedDirs.filter(dir => !DEFAULT_ALLOWED_DIRS.some(d => {
    try {
      return app.getPath(d as any) === dir
    } catch {
      return false
    }
  })))
  
  return true
}

/**
 * 获取所有授权目录
 */
export function getAuthorizedDirs(): string[] {
  return [...allowedDirs]
}
```

```typescript
import { wrapHandler } from './utils'

export function registerFsHandlers(): void {
  ipcMain.handle('fs:readFileAsBuffer', wrapHandler((_event, filePath: string) => {

    const buffer = fs.readFileSync(validPath)
    return buffer.buffer.slice(buffer.byteOffset, buffer.byteOffset + buffer.byteLength)
  }, 'fs:readFileAsBuffer'))
  
  ipcMain.handle('fs:readFileAsText', wrapHandler((_event, filePath: string) => {
    const validPath = validateFilePath(filePath)
    return fs.readFileSync(validPath, { encoding: 'utf-8' })
  }, 'fs:readFileAsText'))
  
  ipcMain.handle('fs:checkFileExists', wrapHandler((_event, filePath: string) => {
    try {
      const validPath = validateFilePath(filePath)
      fs.accessSync(validPath, fs.constants.R_OK)
      return true
    } catch {
      return false
    }
  }, 'fs:checkFileExists'))
}
```

### 1.3 敏感数据加密

#### 完整实现

```typescript
// electron/crypto.ts
import { safeStorage } from 'electron'

/**
 * 检查加密是否可用
 */
export function isEncryptionAvailable(): boolean {
  return safeStorage.isEncryptionAvailable()
}

/**
 * 加密字符串
 * @param plaintext 明文字符串
 * @returns Base64 编码的加密字符串
 */
export function encryptString(plaintext: string): string {
  if (!isEncryptionAvailable()) {
    // 降级: 使用 Base64 编码 (不安全，但至少不是明文)
    console.warn('Encryption not available, falling back to Base64')
    return Buffer.from(plaintext).toString('base64')
  }
  
  const encrypted = safeStorage.encryptString(plaintext)
  return encrypted.toString('base64')
}

/**
 * 解密字符串
 * @param encryptedBase64 Base64 编码的加密字符串
 * @returns 解密后的明文字符串
 */
export function decryptString(encryptedBase64: string): string {
  if (!isEncryptionAvailable()) {
    // 降级: Base64 解码
    return Buffer.from(encryptedBase64, 'base64').toString()
  }
  
  const buffer = Buffer.from(encryptedBase64, 'base64')
  return safeStorage.decryptString(buffer).toString()
}

/**
 * 加密 JSON 对象
 * @param obj 要加密的对象
 * @returns Base64 编码的加密字符串
 */
export function encryptJSON<T>(obj: T): string {
  const plaintext = JSON.stringify(obj)
  return encryptString(plaintext)
}

/**
 * 解密 JSON 对象
 * @param encryptedBase64 Base64 编码的加密字符串
 * @returns 解密后的对象，失败返回 null
 */
export function decryptJSON<T>(encryptedBase64: string): T | null {
  try {
    const plaintext = decryptString(encryptedBase64)
    return JSON.parse(plaintext) as T
  } catch (err) {
    console.error('Failed to decrypt JSON:', err)
    return null
  }
}
```

### 1.4 错误处理

#### 统一错误处理模块

```typescript
// electron/errorHandler.ts
import { app, dialog, shell } from 'electron'
import fs from 'fs'
import path from 'path'

const LOG_FILE = path.join(app.getPath('userData'), 'error.log')
const MAX_LOG_SIZE = 5 * 1024 * 1024 // 5MB

/**
 * 初始化错误处理
 */
export function setupErrorHandler(): void {
  // 主进程未捕获异常
  process.on('uncaughtException', (error) => {
    logError('Uncaught Exception', error)
    showErrorDialog(error)
  })
  
  // Promise 未处理拒绝
  process.on('unhandledRejection', (reason, promise) => {
    logError('Unhandled Rejection', reason)
  })
  
  // GPU 进程崩溃
  app.on('gpu-process-crashed', (event, killed) => {
    logError('GPU Process Crashed', { killed })
  })
}

/**
 * 记录错误日志
 */
export function logError(type: string, error: unknown): void {
  const timestamp = new Date().toISOString()
  const message = error instanceof Error ? error.stack || error.message : String(error)
  const logEntry = `[${timestamp}] ${type}\n${message}\n\n`
  
  try {
    // 检查日志文件大小，超过限制则清空
    if (fs.existsSync(LOG_FILE)) {
      const stat = fs.statSync(LOG_FILE)
      if (stat.size > MAX_LOG_SIZE) {
        fs.writeFileSync(LOG_FILE, '')
      }
    }
    
    fs.appendFileSync(LOG_FILE, logEntry)
  } catch {
    // 日志写入失败，静默处理
  }
  
  // 同时输出到控制台
  console.error(`[${type}]`, error)
}

/**
 * 显示错误对话框
 */
export function showErrorDialog(error: unknown, title = '错误'): void {
  const message = error instanceof Error ? error.message : String(error)
  
  dialog.showMessageBox({
    type: 'error',
    title,
    message: '应用发生错误',
    detail: message,
    buttons: ['确定', '查看日志'],
    defaultId: 0,
  }).then(({ response }) => {
    if (response === 1) {
      shell.openPath(LOG_FILE)
    }
  })
}

/**
 * 获取错误日志路径
 */
export function getErrorLogPath(): string {
  return LOG_FILE
}

/**
 * 清空错误日志
 */
export function clearErrorLog(): void {
  try {
    fs.writeFileSync(LOG_FILE, '')
  } catch {
    // 静默处理
  }
}
```

```typescript
// electron/ipc/utils.ts
import { logError } from '../errorHandler'

/**
 * 包装 IPC 处理器，添加错误处理
 * @param handler 原始处理器函数
 * @param handlerName 处理器名称，用于日志
 * @returns 包装后的处理器
 */
export function wrapHandler<T extends (...args: any[]) => any>(
  handler: T,
  handlerName: string
): T {
  return ((...args: any[]) => {
    try {
      const result = handler(...args)
      
      // 处理异步处理器
      if (result instanceof Promise) {
        return result.catch(err => {
          logError(`IPC Error: ${handlerName}`, err)
          throw err
        })
      }
      
      return result
    } catch (err) {
      logError(`IPC Error: ${handlerName}`, err)
      throw err
    }
  }) as T
}

//   'fs:readFileAsBuffer',
```

### 2.1 EPUB 按需加载

#### 主进程接口

```typescript
// electron/ipc/epub.ts
import { ipcMain } from 'electron'
import AdmZip from 'adm-zip'
import fs from 'fs'
import { wrapHandler } from './utils'

// 缓存 zip 实例，避免重复打开文件
const zipCache = new Map<string, { zip: AdmZip; mtime: number }>()

/**
 * 获取或创建 zip 实例 (带缓存)
 */
function getZip(filePath: string): AdmZip {
  const stat = fs.statSync(filePath)
  const cached = zipCache.get(filePath)
  
  if (cached && cached.mtime === stat.mtimeMs) {
    return cached.zip
  }
  
  const zip = new AdmZip(filePath)
  zipCache.set(filePath, { zip, mtime: stat.mtimeMs })
  return zip
}

export function registerEpubHandlers(): void {
  // 获取 EPUB 文件列表
  ipcMain.handle('epub:listEntries', wrapHandler((_event, filePath: string) => {
    const zip = getZip(filePath)
    const entries = zip.getEntries()
    
    return entries
      .filter(e => !e.isDirectory)
      .map(e => ({
        name: e.entryName,
        size: e.header.size,
        compressedSize: e.header.compressedSize,
        time: e.header.time,
      }))
  }, 'epub:listEntries'))
  
  // 按需获取单个文件
  ipcMain.handle('epub:getEntry', wrapHandler((_event, filePath: string, entryName: string) => {
    const zip = getZip(filePath)
    const entry = zip.getEntry(entryName)
    
    if (!entry) return null
    
    const data = entry.getData()
    return {
      name: entry.entryName,
      data: Array.from(data),
      size: data.length,
    }
  }, 'epub:getEntry'))
  
  // 获取多个文件 (批量)
  ipcMain.handle('epub:getEntries', wrapHandler((_event, filePath: string, entryNames: string[]) => {
    const zip = getZip(filePath)
    
    return entryNames.map(name => {
      const entry = zip.getEntry(name)
      if (!entry) return { name, data: null }
      
      const data = entry.getData()
      return {
        name: entry.entryName,
        data: Array.from(data),
        size: data.length,
      }
    })
  }, 'epub:getEntries'))
  
  // 清理缓存
  ipcMain.handle('epub:clearCache', wrapHandler((_event, filePath?: string) => {
    if (filePath) {
      zipCache.delete(filePath)
    } else {
      zipCache.clear()
    }
  }, 'epub:clearCache'))
}
```

#### 渲染进程服务

```typescript
// src/services/epubService.ts

interface EpubEntry {
  name: string
  size: number
  compressedSize: number
  time: Date
}

interface EpubEntryData {
  name: string
  data: number[] | null
  size: number
}

/**
 * EPUB 按需加载服务
 */
export class EpubService {
  private filePath: string
  private entryCache = new Map<string, Uint8Array>()
  private entryList: EpubEntry[] | null = null
  private maxCacheSize = 50 * 1024 * 1024 // 50MB
  private currentCacheSize = 0
  
  constructor(filePath: string) {
    this.filePath = filePath
  }
  
  /**
   * 获取文件列表
   */
  async listEntries(): Promise<EpubEntry[]> {
    if (!this.entryList) {
      this.entryList = await window.electronAPI.epub.listEntries(this.filePath)
    }
    return this.entryList
  }
  
  /**
   * 获取单个文件
   * @param entryName 文件路径
   * @returns 文件数据，不存在返回 null
   */
  async getEntry(entryName: string): Promise<Uint8Array | null> {
    // 检查缓存
    const cached = this.entryCache.get(entryName)
    if (cached) {
      return cached
    }
    
    // 从主进程加载
    const entry = await window.electronAPI.epub.getEntry(this.filePath, entryName)
    if (!entry) return null
    
    const data = new Uint8Array(entry.data)
    
    // 添加到缓存
    this.addToCache(entryName, data)
    
    return data
  }
  
  /**
   * 批量获取文件
   * @param entryNames 文件路径列表
   * @returns 文件数据 Map
   */
  async getEntries(entryNames: string[]): Promise<Map<string, Uint8Array>> {
    const result = new Map<string, Uint8Array>()
    const toLoad: string[] = []
    
    // 检查缓存
    for (const name of entryNames) {
      const cached = this.entryCache.get(name)
      if (cached) {
        result.set(name, cached)
      } else {
        toLoad.push(name)
      }
    }
    
    // 批量加载未缓存的文件
    if (toLoad.length > 0) {
      const entries = await window.electronAPI.epub.getEntries(this.filePath, toLoad)
      for (const entry of entries) {
        if (entry.data) {
          const data = new Uint8Array(entry.data)
          this.addToCache(entry.name, data)
          result.set(entry.name, data)
        }
      }
    }
    
    return result
  }
  
  /**
   * 预加载文件
   * @param entryNames 要预加载的文件路径列表
   */
  async preload(entryNames: string[]): Promise<void> {
    await this.getEntries(entryNames)
  }
  
  /**
   * 添加到缓存 (LRU 策略)
   */
  private addToCache(name: string, data: Uint8Array): void {
    // 如果数据太大，不缓存
    if (data.length > this.maxCacheSize / 2) {
      return
    }
    
    // 清理缓存直到有足够空间
    while (this.currentCacheSize + data.length > this.maxCacheSize && this.entryCache.size > 0) {
      const firstKey = this.entryCache.keys().next().value
      if (firstKey) {
        const removed = this.entryCache.get(firstKey)
        if (removed) {
          this.currentCacheSize -= removed.length
        }
        this.entryCache.delete(firstKey)
      }
    }
    
    this.entryCache.set(name, data)
    this.currentCacheSize += data.length
  }
  
  /**
   * 清理缓存
   */
  clearCache(): void {
    this.entryCache.clear()
    this.currentCacheSize = 0
    window.electronAPI.epub.clearCache(this.filePath)
  }
  
  /**
   * 销毁服务
   */
  destroy(): void {
    this.clearCache()
    this.entryList = null
  }
}

// 工厂函数
const serviceCache = new Map<string, EpubService>()

/**
 * 获取 EPUB 服务实例
 */
export function getEpubService(filePath: string): EpubService {
  if (!serviceCache.has(filePath)) {
    serviceCache.set(filePath, new EpubService(filePath))
  }
  return serviceCache.get(filePath)!
}

/**
 * 销毁 EPUB 服务实例
 */
export function destroyEpubService(filePath: string): void {
  const service = serviceCache.get(filePath)
  if (service) {
    service.destroy()
    serviceCache.delete(filePath)
  }
}
```

```typescript
// src/services/ttsPlayer.ts
export class TTSPlayer {
  private audioResolvers = new Map<number, {
    resolve: (data: ArrayBuffer) => void
    reject: (err: Error) => void
    timeout: NodeJS.Timeout
  }>()
  
  /**
   * 等待音频数据 (替代轮询)
   * @param index 音频索引
   * @param timeoutMs 超时时间 (毫秒)
   * @returns 音频数据
   */
  private waitForAudio(index: number, timeoutMs = 30000): Promise<ArrayBuffer> {
    // 检查是否已有数据
    const existing = this.audioQueue[index]
    if (existing) {
      return Promise.resolve(existing)
    }
    
    return new Promise((resolve, reject) => {
      // 设置超时
      const timeout = setTimeout(() => {
        this.audioResolvers.delete(index)
        reject(new Error(`Audio wait timeout for index ${index}`))
      }, timeoutMs)
      
      this.audioResolvers.set(index, { resolve, reject, timeout })
    })
  }
  
  /**
   * 当音频数据准备好时调用
   * @param index 音频索引
   * @param data 音频数据
   */
  private onAudioReady(index: number, data: ArrayBuffer): void {
    const resolver = this.audioResolvers.get(index)
    if (resolver) {
      clearTimeout(resolver.timeout)
      this.audioResolvers.delete(index)
      resolver.resolve(data)
    }
  }
  
  /**
   * 当音频加载失败时调用
   * @param index 音频索引
   * @param error 错误信息
   */
  private onAudioError(index: number, error: Error): void {
    const resolver = this.audioResolvers.get(index)
    if (resolver) {
      clearTimeout(resolver.timeout)
      this.audioResolvers.delete(index)
      resolver.reject(error)
    }
  }
  
  /**
   * 清理所有等待
   */
  private clearAllWaiters(): void {
    for (const [index, resolver] of this.audioResolvers) {
      clearTimeout(resolver.timeout)
      resolver.reject(new Error('Player stopped'))
    }
    this.audioResolvers.clear()
  }
  
  /**
   * 停止播放
   */
  stop(): void {
    this.clearAllWaiters()
    // ... 其他停止逻辑
  }
}
```

```typescript
import { ref, computed } from 'vue'
import { useReaderStore } from '@/stores/reader'
import { useLibraryStore } from '@/stores/library'
import { getCoverUrl, revokeCoverUrl } from '@/utils/cover'
import { useRouter } from 'vue-router'

export function useBookAnimation() {
  const router = useRouter()
  const reader = useReaderStore()
  const library = useLibraryStore()
  
  // 动画状态
  const animating = ref(false)
  const animDirection = ref<'open' | 'close'>('open')
  const animPhase = ref<'fly' | 'open' | 'fade' | 'unfade' | 'close-book' | 'fly-back'>('fly')
  
  // 动画数据
  const animCoverUrl = ref('')
  const animCoverKey = ref('')
  const animRect = ref<{ left: number; top: number; width: number; height: number } | null>(null)
  const animBookTitle = ref('')
  
  // 计算样式
  const flyStyle = computed(() => {
    const r = animRect.value
    return {
      '--start-x': (r ? r.left + r.width / 2 : 400) + 'px',
      '--start-y': (r ? r.top + r.height / 2 : 350) + 'px',
      '--start-w': (r ? r.width : 130) + 'px',
      '--start-h': (r ? r.height : 175) + 'px',
    }
  })
  
  // 辅助函数
  function delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms))
  }
  
  // 打开书籍动画
  async function startBookOpenAnimation(
    bookId: string,
    coverUrl: string,
    rect: DOMRect,
    title: string
  ): Promise<void> {
    const book = library.books.find((b) => b.id === bookId)
    if (!book || book.invalid) return
    
    // 保存动画数据
    animCoverKey.value = book.coverKey || ''
    revokeCoverUrl(animCoverUrl.value)
    animCoverUrl.value = book.coverKey ? await getCoverUrl(book.coverKey) : ''
    animRect.value = {
      left: rect.left,
      top: rect.top,
      width: rect.width,
      height: rect.height,
    }
    animBookTitle.value = title
    
    // 开始动画
    animating.value = true
    animDirection.value = 'open'
    animPhase.value = 'fly'
    
    // 打开书籍
    reader.openBook(book)
    
    // Phase 1: 飞到中央
    await delay(260)
    animPhase.value = 'open'
    
    // Phase 2: 翻书展开
    await delay(340)
    animPhase.value = 'fade'
    
    // Phase 3: 淡入阅读器
    await delay(220)
    
    // 动画完成
    animating.value = false
    revokeCoverUrl(animCoverUrl.value)
    animCoverUrl.value = ''
    
    // 导航到阅读器
    router.push({ name: 'reader', params: { bookId } })
  }
  
  // 关闭书籍动画
  async function startBookCloseAnimation(): Promise<void> {
    if (!reader.currentBook) return
    
    // 保存进度
    reader.saveProgress()
    
    // 准备封面
    revokeCoverUrl(animCoverUrl.value)
    animCoverUrl.value = animCoverKey.value ? await getCoverUrl(animCoverKey.value) : ''
    
    // 开始动画
    animating.value = true
    animDirection.value = 'close'
    
    // Phase 1: 阅读器淡出
    animPhase.value = 'unfade'
    await delay(200)
    
    // 关闭书籍
    reader.closeBook()
    
    // Phase 2: 合书
    animPhase.value = 'close-book'
    await delay(320)
    
    // Phase 3: 封面飞回
    animPhase.value = 'fly-back'
    await delay(270)
    
    // 动画完成
    animating.value = false
    revokeCoverUrl(animCoverUrl.value)
    animCoverUrl.value = ''
    
    // 重新加载书架
    library.loadBooks()
    
    // 导航到书架
    router.push({ name: 'bookshelf' })
  }
  
  return {
    animating,
    animDirection,
    animPhase,
    animCoverUrl,
    animBookTitle,
    flyStyle,
    startBookOpenAnimation,
    startBookCloseAnimation,
  }
}
```

```typescript
const BOOK_EXTS = new Set(['.epub', '.txt', '.mobi', '.azw3', '.cbz', '.cbr'])

export function useExternalFileOpen() {
  const router = useRouter()
  const appModeStore = useAppModeStore()
  const library = useLibraryStore()
  const reader = useReaderStore()
  
  let cleanup: (() => void) | null = null
  
  async function handleExternalFileOpen(filePath: string): Promise<void> {
    if (!filePath) return
    
    const ext = filePath.substring(filePath.lastIndexOf('.')).toLowerCase()
    
    if (ext === '.md') {
      // Markdown 文件 -> 简记模式
      if (appModeStore.isReaderMode) {
        appModeStore.switchToNote()
      }
      router.push({ name: 'note' })
      // 等待路由切换完成后再打开文件
      setTimeout(() => {
        window.dispatchEvent(new CustomEvent('note-open-external', {
          detail: { filePath }
        }))
      }, 100)
    } else if (BOOK_EXTS.has(ext)) {
      // 书籍文件 -> 简阅模式
      if (appModeStore.isNoteMode) {
        appModeStore.switchToReader()
      }
      
      try {
        await library.importBook([filePath])
        const { hashPath } = await import('@/services/fileService')
        const bookId = hashPath(filePath)
        const book = library.books.find(b => b.id === bookId)
        
        if (book && !book.invalid) {
          reader.openBook(book)
          router.push({ name: 'reader', params: { bookId } })
        }
      } catch (err) {
        console.error('Failed to open external book:', err)
      }
    }
  }
  
  onMounted(() => {
    // 监听后续文件打开事件
    if (window.electronAPI?.onFileOpen) {
      cleanup = window.electronAPI.onFileOpen(handleExternalFileOpen)
    }
    
    // 首次启动：获取待处理的文件
    if (window.electronAPI?.app?.getPendingFilePath) {
      window.electronAPI.app.getPendingFilePath().then((fp: string | null) => {
        if (fp) handleExternalFileOpen(fp)
      })
    }
  })
  
  onBeforeUnmount(() => {
    cleanup?.()
  })
  
  return {
    handleExternalFileOpen,
  }
}
```

### 4.3 文件系统 IPC

```typescript
// electron/ipc/fs.ts
import { ipcMain } from 'electron'
import fs from 'fs'
import path from 'path'
import { validateFilePath } from '../security'
import { wrapHandler } from './utils'

export function registerFsHandlers(): void {
  ipcMain.handle('fs:readFileAsBuffer', wrapHandler((_event, filePath: string) => {
    const validPath = validateFilePath(filePath)
    const buffer = fs.readFileSync(validPath)
    return buffer.buffer.slice(buffer.byteOffset, buffer.byteOffset + buffer.byteLength)
  }, 'fs:readFileAsBuffer'))
  
  ipcMain.handle('fs:readFileAsText', wrapHandler((_event, filePath: string) => {
    const validPath = validateFilePath(filePath)
    return fs.readFileSync(validPath, { encoding: 'utf-8' })
  }, 'fs:readFileAsText'))
  
  ipcMain.handle('fs:checkFileExists', wrapHandler((_event, filePath: string) => {
    try {
      const validPath = validateFilePath(filePath)
      fs.accessSync(validPath, fs.constants.R_OK)
      return true
    } catch {
      return false
    }
  }, 'fs:checkFileExists'))
  
  ipcMain.handle('fs:getFileName', wrapHandler((_event, filePath: string) => {
    return path.basename(filePath)
  }, 'fs:getFileName'))
  
  ipcMain.handle('fs:getFileSize', wrapHandler((_event, filePath: string) => {
    const validPath = validateFilePath(filePath)
    const stat = fs.statSync(validPath)
    return stat.size
  }, 'fs:getFileSize'))
  
  ipcMain.handle('fs:copyFile', wrapHandler((_event, src: string, dest: string) => {
    const validSrc = validateFilePath(src)
    fs.copyFileSync(validSrc, dest)
  }, 'fs:copyFile'))
  
  ipcMain.handle('fs:deleteFile', wrapHandler((_event, filePath: string) => {
    const validPath = validateFilePath(filePath)
    fs.unlinkSync(validPath)
  }, 'fs:deleteFile'))
  
  ipcMain.handle('fs:ensureDir', wrapHandler((_event, dirPath: string) => {
    fs.mkdirSync(dirPath, { recursive: true })
  }, 'fs:ensureDir'))
  
  ipcMain.handle('fs:readDir', wrapHandler((_event, dirPath: string) => {
    return fs.readdirSync(dirPath)
  }, 'fs:readDir'))
}
```
